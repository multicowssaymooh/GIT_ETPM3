366 /** ***************************************************************************
367  * @brief Check for selection/transition
368  *
369  * If the last transition has been consumed (MENU_NONE == MENU_transition)
370  * and the touchscreen has been touched for a defined period
371  * the variable MENU_transition is set to the touched item.
372  *****************************************************************************/
373 void MENU_check_transition(void)
374 {
375  static MENU_item_t item_old = MENU_NONE;
376  static MENU_item_t item_new = MENU_NONE;
377  static TS_StateTypeDef TS_State; // State of the touch controller
378  BSP_TS_GetState(&TS_State); // Get the state
379  if (TS_State.TouchDetected) { // If a touch was detected
380  /* Do only if last transition not pending anymore */
381  if (MENU_NONE == MENU_transition) {
382  item_old = item_new; // Store old item
383  /* If touched within the menu bar? */
384  if ((MENU_Y < TS_State.Y) && (MENU_Y+MENU_HEIGHT > TS_State.Y)) {
385  item_new = TS_State.X // Calculate new item
386  / (BSP_LCD_GetXSize()/MENU_ENTRY_COUNT);
387  if ((0 > item_new) || (MENU_ENTRY_COUNT <= item_new)) {
388  item_new = MENU_NONE; // Out of bounds
389  }
390  if (item_new == item_old) { // 2 times the same menu item
391  item_new = MENU_NONE;
392  MENU_transition = item_old;
393  }
394  }
395  }
396  }
397 }

401 /** ***************************************************************************
402  * @brief Interrupt handler for the touchscreen
403  *
404  * @note BSP_TS_ITConfig(); must be called in the main function
405  * to enable touchscreen interrupt.
406  * @note There are timing issues when interrupt is enabled.
407  * It seems that polling is the better choice with this evaluation board.
408  * @n Call MENU_check_transition() from the while loop in main for polling.
409  *
410  * The touchscreen interrupt is connected to PA15.
411  * @n The interrupt handler for external line 15 to 10 is called.
412  *****************************************************************************/
413 void EXTI15_10_IRQHandler(void)
414 {
415  if (EXTI->PR & EXTI_PR_PR15) { // Check if interrupt on touchscreen
416  EXTI->PR |= EXTI_PR_PR15; // Clear pending interrupt on line 15
417  if (BSP_TS_ITGetStatus()) { // Get interrupt status
418  BSP_TS_ITClear(); // Clear touchscreen controller int.
419  MENU_check_transition();
420  }
421  EXTI->PR |= EXTI_PR_PR15; // Clear pending interrupt on line 15
422  }
423 }
