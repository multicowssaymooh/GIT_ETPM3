317 void gyro_disable(void)
318 {
319  __HAL_RCC_GPIOC_CLK_ENABLE(); // Enable Clock for GPIO port C
320  /* Disable PC1 and PF8 first */
321  GPIOC->MODER &= ~GPIO_MODER_MODER1; // Reset mode for PC1
322  GPIOC->MODER |= GPIO_MODER_MODER1_0; // Set PC1 as output
323  GPIOC->BSRR |= GPIO_BSRR_BR1; // Set GYRO (CS) to 0 for a short time
324  Delay_us(1); // Wait some time
325  GPIOC->MODER |= GPIO_MODER_MODER1_Msk; // Analog mode PC1 = ADC123_IN11
326  __HAL_RCC_GPIOF_CLK_ENABLE(); // Enable Clock for GPIO port F
327  GPIOF->OSPEEDR &= ~GPIO_OSPEEDR_OSPEED8; // Reset speed of PF8
328  GPIOF->AFR[1] &= ~GPIO_AFRH_AFSEL8; // Reset alternate func. of PF8
329  GPIOF->PUPDR &= ~GPIO_PUPDR_PUPD8; // Reset pulup/down of PF8
330  Delay_us(1); // Wait some time
